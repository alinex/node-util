¶ Filter object

This will create a new object with all keys that pass the test implemented by the provided allow(value, key, object)

 
filter = exports.filter = (obj, allow) ->
  result = {}
  for key of obj
    if obj.hasOwnProperty(key) and allow obj[key], key, obj
      result[key] = obj[key]
  result
